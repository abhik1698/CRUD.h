Create Table Syntax

MySQL syntax:

CREATE TABLE tableName( id INT, name VARCHAR(40) );

MySQL Shorthand syntax:

c tableName id name

Drop Table Syntax

MySQL:

DROP TABLE tableName;

MySQL Shorthand:

dr tableName

Drop Schema Syntax

MySQL:

DROP DATABASE databaseName;

MySQL Shorthand:

r

Insert Row Syntax

MySQL:

INSERT INTO tableName VALUES( 1, 'name' );

MySQL Shorthand:

i tableName 1 name

Delete Row Syntax

MySQL:

DELETE FROM tableName WHERE id = 1;

MySQL Shorthand:

d tableName id 1

Truncate Table Syntax

MySQL:

TRUNCATE TABLE tableName;

MySQL Shorthand:

t tableName
